const startDate = new Date("2026-01-31T17:43:00"); const milestones = [ { name: "Day 100", daysAfterStart: 100 }, { name: "Day 150", daysAfterStart: 150 }, { name: "Day 200", daysAfterStart: 200 }, { name: "Anniversary", daysAfterStart: 365 } ]; function updateMilestones() { const now = new Date(); milestones.forEach(m => { const milestoneDate = new Date(startDate.getTime() + m.daysAfterStart*24*60*60*1000); const diff = milestoneDate - now; const el = document.getElementById(m.name.replace(" ", "_")); if (!el) return; if (diff <= 0) { el.innerHTML = m.name + " ✅ Done!"; } else { const days = Math.floor(diff / (1000*60*60*24)); const hours = Math.floor((diff / (1000*60*60)) % 24); const minutes = Math.floor((diff / (1000*60)) % 60); const seconds = Math.floor((diff / 1000) % 60); el.innerHTML = `${m.name}: ${days}d ${hours}h ${minutes}m ${seconds}s`; } }); } setInterval(updateMilestones, 1000); updateMilestones();
